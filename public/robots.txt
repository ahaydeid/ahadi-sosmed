import { MetadataRoute } from "next"

export default function robots(): MetadataRoute.Robots {
  const base = "https://ahadi.my.id"
  return {
    rules: {
      userAgent: "*",
      allow: "/",
      disallow: ["/api/", "/auth/", "/app/", "/login", "/signup", "/profile/"],
    },
    sitemap: `${base}/sitemap.xml`,
    host: base,
  }
}
